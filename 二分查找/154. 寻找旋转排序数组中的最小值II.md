有重复数字在里面怎么办？
```code
var findMin = function(nums) {
let left = 0;
let right = nums.length-1;
while(left<right){
    let mid = left + Math.floor((right-left)/2);
    if(nums[mid]<nums[right]){
        right = mid;
    }
    else if(nums[mid] > nums[right]){
        left = mid+1;
    }
    else{
        right-=1
    }
}
return nums[left];
};
```
