map只存第一次reminder出现的索引值。     

```code
var checkSubarraySum = function(nums, k) {
let map = new Map();
let sum=0;
map.set(0,-1);
for(let i=0;i<nums.length;i++){
    sum+=nums[i];
    let reminder = ((sum+k) % k+k)%k;
    if (map.has(reminder)) {
        if (i - map.get(reminder) > 1) {
            return true;
        }
    } 
    else{
            map.set(reminder, i);
        }
}
return false;
};
```
