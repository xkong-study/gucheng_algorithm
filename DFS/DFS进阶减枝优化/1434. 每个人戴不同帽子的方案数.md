这道题是针对优化的最后一个方法。   

```code
var numberWays = function(hats) {
let set = new Set();
let count = 0;
function dfs(index){
    if(index == hats.length ){
       count++;
       return;
    }
    let len = hats[index].length;
    for(let i=0;i<len;i++){
        if(set.has(hats[index][i])) continue;
        set.add(hats[index][i]);
        dfs(index+1)
        set.delete(hats[index][i]);
    }
}
dfs(0);
return count;
};
```
