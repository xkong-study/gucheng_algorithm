var ladderLength = function(beginWord, endWord, wordList) {
let queue = []
const wordSet = new Set(wordList);
queue.push([beginWord,1]);
let level = 1;
while(queue.length){
   const [word,level] = queue.shift();
   if(word == endWord) {
       return level;
   }
   for(let i=0;i<word.length;i++){
      for(let j=97;j<=122;j++){
          let ans = word.slice(0,i)+String.fromCharCode(j)+word.slice(i+1);
          if(wordSet.has(ans)){
              queue.push([ans,level+1]);
              wordSet.delete(ans);
          }
      }
   }
}
return 0;
};
