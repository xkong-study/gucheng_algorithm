var reverseBetween = function(head, left, right) {
let dummy = new ListNode(null,head);
let prev = dummy;
for(let i=0;i<left-1;i++){
    prev = prev.next;
}
let Next = prev.next;
let cur = prev.next;
let pre = prev;
for(let i=0;i<=right-left;i++){
    let next = cur.next;
    cur.next = pre;
    pre = cur;
    cur = next;
}
Next.next = cur;
prev.next = pre;
return dummy.next;
};
