var addTwoNumbers = function(l1, l2) {
let count = 0;
let sum = new ListNode(0);
let dummy = sum;
while(l1 || l2 || count){
    let add = count;
    if(l1){
        add+=l1.val;
        l1 = l1.next;
    }
    if(l2){
        add+=l2.val;
        l2 = l2.next;
    }
    sum.next = new ListNode(add%10);
    count = Math.floor(add/10);
    sum = sum.next;
}
return dummy.next;
};
