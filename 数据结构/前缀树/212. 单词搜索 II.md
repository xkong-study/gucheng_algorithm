把每一个格子作为起点然后上下左右寻找有没有满足条件的情况。 dfs(trie.children, i, j, '');          
trie.children：传入 Trie 树的根节点的子节点，这是因为 Trie 树的根节点自身通常不存储字符，而其子节点存储了第一层的所有可能字符。     

```code
var findWords = function(board, words) {
let trie = new Trie();
let result = new Set();
for(let word of words){
    trie.insert(word);
};
let width = board[0].length;
let height = board.length;
function dfs(node,i,j,path){
    if(node.isEnd == true){
        result.add(path);
    }
    if(i<0 || j<0 || i>=height || j>=width || board[i][j]=='£') return;
    let char = board[i][j];
    node = node[char];
    board[i][j] = '£';
    if(!node){
    board[i][j] = char;
    return 
    }

    dfs(node,i+1,j,path+char);
    dfs(node,i-1,j,path+char);
    dfs(node,i,j+1,path+char);
    dfs(node,i,j-1,path+char);
    board[i][j] = char;
}
for (let i = 0; i < height; i++) {
    for (let j = 0; j < width; j++) {
        dfs(trie.children, i, j, '');
    }
}
return Array.from(result);
};


var Trie = function(){
    this.children = {};
}

Trie.prototype.insert = function(word){
    let node = this.children;
    for(const cn of word){
        if(!node[cn]){
            node[cn] = {};
        }
        node = node[cn];
    }
    node.isEnd = true;
}

Trie.prototype.search = function(prefix){
    let node = this.children;
    for(const cn of prefix){
        if(!node[cn]){
           return false;
        }
        node = node[cn];
    }
if(node.isEnd==true) return node;
}
```
