反向思路：
找能独立的区间数量，用总长减去独立的区间数量。排序
```code
var eraseOverlapIntervals = function(intervals) {
if (!intervals.length) {
    return 0;
}
intervals.sort((a,b)=>a[1]-b[1]);
let pre = intervals[0];
let count = 1;
for(let i=1;i<intervals.length;i++){
    if(pre[1]<=intervals[i][0]){
        count++;
        pre = intervals[i];
    }
}
return intervals.length - count;
};
```
