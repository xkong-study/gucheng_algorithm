计数排序        
```code
var sortColors = function(nums) {
let count = new Array(3).fill(0);
let index = 0;
for(let i=0;i<nums.length;i++){
    count[nums[i]]+=1;
}
for(let i=0;i<3;i++){
    while(count[i]>0){
        nums[index++] = i;
        count[i]-=1;
    }
}
return nums;
};
```
