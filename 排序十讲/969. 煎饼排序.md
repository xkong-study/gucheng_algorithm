var pancakeSort = function(arr) {
    const flips = [];
    for (let n = arr.length; n > 1; n--) {
        let index = 0;
        for (let i = 1; i < n; i++) {
            if (arr[i] > arr[index]) {
                index = i;
            }
        }
        if (index === n - 1) continue;
        if (index > 0) {
            reverse(arr, index);
            flips.push(index + 1);
        }
        reverse(arr, n - 1);
        flips.push(n);
    }
    return flips;
};

function reverse(arr, k) {
    let left = 0, right = k;
    while (left < right) {
        let temp = arr[left];
        arr[left++] = arr[right];
        arr[right--] = temp;
    }
}
