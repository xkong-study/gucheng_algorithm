var pancakeSort = function(arr) {
    let ans = [];
    for (let count = arr.length; count > 0; count--) {
        let index = arr.indexOf(Math.max(...arr.slice(0, count)));
        if (index === count - 1) continue;
        if (index > 0) {
            reverse(arr, index);
            ans.push(index + 1);
        }
        reverse(arr, count - 1);
        ans.push(count);
    }
    return ans;
};

function reverse(arr, n) {
    let left = 0, right = n;
    while (left < right) {
        [arr[left], arr[right]] = [arr[right], arr[left]];
        left++;
        right--;
    }
}
